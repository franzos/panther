 (entry (commit "00a45578dc4ea5989503954ca8ccc8efb0fe9717")
        (title
         (en "Substitute servers: nonguix routed through guix.moe mirror"))
        (body
         (en "@code{substitutes.nonguix.org} has been unreliable, so it has
been replaced with @code{cache-test.guix.moe}, a transparent mirror that
retains the original nonguix signing key.  No action is required: the
nonguix key remains authorized.

Additionally, @code{cache-fi.guix.moe} (the guix.moe build farm head node)
has been added as a secondary mirror with its own signing key.

After @code{guix pull}, run @code{sudo guix system reconfigure} to apply
the new substitute URLs and authorized keys.")))

 (entry (commit "aa2e3a89aea0e818d51183267a28010cf53ce694")
        (title
         (en "System modules reorganized"))
        (body
         (en "The @code{(px system panther)} module has been reorganized into
@code{(px system common)} and @code{(px system os)}.

All @code{%panther-*} variables in @code{(px system panther)} are now
deprecated aliases.  Update your imports:

@itemize
@item
@code{%panther-base-services} @result{} @code{%os-base-services}
@item
@code{%panther-base-packages} @result{} @code{%os-base-packages}
@item
@code{%panther-desktop-services-minimal} @result{} @code{%os-desktop-services-minimal}
@item
@code{%panther-os} @result{} @code{%os-base}
@end itemize

Import @code{(px system os)} instead of @code{(px system panther)}.")))

 (entry (commit "4f51b01")

        (body
         (en "The @code{home-darkman-service-type} has been removed from this

in your imports.

@item
Change @code{(use-geoclue? #f)} to @code{(use-geoclue #f)} in your
@code{home-darkman-configuration}.
@end enumerate

The upstream service is functionally equivalent."))))
